<script>

let money=0
let clickPower=1
let autoPower=0

let inventory=[]

const icons=[

{name:"ネコ",url:"https://cdn-icons-png.flaticon.com/512/616/616408.png",rate:40,rarity:"N"},

{name:"犬",url:"https://cdn-icons-png.flaticon.com/512/616/616430.png",rate:30,rarity:"R"},

{name:"ドラゴン",url:"https://cdn-icons-png.flaticon.com/512/616/616554.png",rate:15,rarity:"SR"},

{name:"王冠",url:"https://cdn-icons-png.flaticon.com/512/616/616490.png",rate:10,rarity:"SSR"},

{name:"神",url:"https://cdn-icons-png.flaticon.com/512/616/616494.png",rate:5,rarity:"UR"}

]

const shop=[

{name:"クリック強化",cost:50,action:()=>{clickPower++}},

{name:"クリック強化Ⅱ",cost:200,action:()=>{clickPower+=5}},

{name:"オートクリック",cost:150,action:()=>{autoPower++}},

{name:"高速オート",cost:800,action:()=>{autoPower+=5}}

]

function clickCoin(){

money+=clickPower

update()

}

function gacha(){

if(money<200){

alert("コインが足りません")

return

}

money-=200

let r=Math.random()*100
let sum=0

for(let item of icons){

sum+=item.rate

if(r<=sum){

inventory.push(item)

showResult(item)

updateInventory()

break

}

}

update()

}

function showResult(item){

let text=item.name+" GET!"

if(item.rarity=="R") text="<span class='R'>"+text+"</span>"
if(item.rarity=="SR") text="<span class='SR'>"+text+"</span>"
if(item.rarity=="SSR") text="<span class='SSR'>"+text+"</span>"
if(item.rarity=="UR") text="<span class='UR'>"+text+"</span>"

document.getElementById("result").innerHTML=text

}

function update(){

document.getElementById("money").innerText=Math.floor(money)

save()

}

function updateInventory(){

let div=document.getElementById("inventory")

div.innerHTML=""

inventory.forEach(item=>{

let img=document.createElement("img")

img.src=item.url

img.onclick=()=>{

document.getElementById("icon").src=item.url

}

div.appendChild(img)

})

save()

}

function createShop(){

let s=document.getElementById("shop")

s.innerHTML=""

shop.forEach(item=>{

let div=document.createElement("div")

div.className="shopItem"

let btn=document.createElement("button")

btn.innerText=item.name+" ("+item.cost+")"

btn.onclick=()=>{

if(money>=item.cost){

money-=item.cost

item.action()

update()

}

}

div.appendChild(btn)

s.appendChild(div)

})

}

function autoLoop(){

money+=autoPower

update()

}

setInterval(autoLoop,1000)

function save(){

localStorage.setItem("money",money)
localStorage.setItem("inventory",JSON.stringify(inventory))
localStorage.setItem("clickPower",clickPower)
localStorage.setItem("autoPower",autoPower)

}

function load(){

money=Number(localStorage.getItem("money"))||0
inventory=JSON.parse(localStorage.getItem("inventory"))||[]
clickPower=Number(localStorage.getItem("clickPower"))||1
autoPower=Number(localStorage.getItem("autoPower"))||0

update()
updateInventory()

}

function resetGame(){

if(confirm("本当にリセットしますか？")){

localStorage.clear()

location.reload()

}

}

createShop()
load()

</script>
